class differentiator():
    def __init__(self,function):
        self.function=function
        self.epsilon=pow(10,-8)
    def h_calc(self,x):
        if abs(x)>=1:
            self.h=(abs(x)*self.epsilon)
        else:
            self.h=self.epsilon
    def one_sided(self,point,h):
        result=self.function(point+h)-self.function(point)
        result=result/h
        return(result)
    def two_sided(self,point,h):
        result=self.function(point+h)-self.function(point-h)
        result=result/(2*h)
        return(result)

Question 3

Solving the maximization problem for the youngsters who live alone:

1) Substituting the expression for consumption into utility:

$\frac{(w\epsilon h_a)^{1-\sigma}}{1-\sigma}-\psi \frac{h_A^{1+1/v}}{1+1/v}$
2) Taking the frist derivative:
$\frac{\partial U}{\partial h_a}=w\epsilon (w\epsilon h_A)^{-\sigma}-\psi h_A^{1/v}=(w\epsilon)^{1-\sigma}h_a^{-\sigma}-\psi h_A^{1/v}=0$
$(w\epsilon)^{1-\sigma}h_a^{-\sigma}=\psi h_A^{1/v}$

$h_A^{-\sigma -1/v}=\frac{\psi}{(w\epsilon)^{1-\sigma}}$
$h_A=(\frac{\psi}{(w\epsilon)^{1-\sigma}})^{\frac{1}{-\sigma -1/v}}$

Solving the problem for the youngsters living with parents:
1) Expressing the utility as Taylor expansion around $h_A$:
$U(h_T)=U(h_A)+U^{\prime}(h_A)(h_T-h_A)+U^{\prime \prime}(h_A)(h_T-h_A)^{2}$
2) Taking the first derivative of the approximation: $\frac{\partial U}{\partial h_T}=U^{\prime}(h_A)+2U^{\prime \prime}(h_A)(h_T-h_A)=0$ $h_T=h_A-\frac{U^{\prime}(h_A)}{2U^{\prime \prime}(h_A)}$
3) Finding the values for $U^{\prime}(h_A)$ and $U^{\prime \prime}(h_A)$:
$U^{\prime}(h_A)=w\epsilon (w\epsilon h_A+\zeta)^{-\sigma}-\psi h_A^{1/v}$
$U^{\prime \prime}(h_A)=-\sigma(w\epsilon)^2 (w\epsilon h_A+\zeta)^{-\sigma-1}-\frac{\psi}{v} h_A^{1/v-1}$
Given this information, I can substitute the expressions into the utility function to obtain $\mu^{*}(\epsilon)$

The basic idea in my code is to use the following transformation:
$\int^{\inf}_{0}f(x)dx=\int^{\inf}_{0}f(x)e^{x}e^{-x}dx=\int^{\inf}_{0}g(x)e^{-x}dx$ And use the Gauss- quadrature to evaluate the integral. Also I will use the idea that the integral inside the double integral is a normal CDF with $\mu^{*}(x)$ as its argument.

def h_a(psi,w,e,sigma,v,zeta):
    return(pow(psi/pow((w*e),(1-sigma)),1/(-sigma-1/v)))
def u_t_prime(psi,w,e,sigma,v,zeta):
    h=h_a(psi,w,e,sigma,v,zeta)
    result=w*e*pow(w*e*h+zeta,0-sigma)-pow(h,(1/v))*psi
    return(result)
def u_t_2prime(psi,w,e,sigma,v,zeta):
    h=h_a(psi,w,e,sigma,v,zeta)
    result=-sigma*pow(w*e,2)*pow(w*e*h+zeta,-sigma-1)-pow(h,(1/v)-1)*psi/v
    return(result)
def h_t(psi,w,e,sigma,v,zeta):
    h=h_a(psi,w,e,sigma,v,zeta)
    u_pr=u_t_prime(psi,w,e,sigma,v,zeta)
    u_2pr=u_t_2prime(psi,w,e,sigma,v,zeta)
    return(h-u_pr/(2*u_2pr))
def u_a(psi,w,e,sigma,v,zeta):
    h=h_a(psi,w,e,sigma,v,zeta)
    c=w*e*h
    result=pow(c,1-sigma)/(1-sigma)-psi*pow(h,1+1/v)/(1+1/v)
    return(result)
    
    
def u_t(psi,w,e,sigma,v,zeta):
    h=h_t(psi,w,e,sigma,v,zeta)
    c=w*e*h
    result=pow(c+zeta,1-sigma)/(1-sigma)-psi*pow(h,1+1/v)/(1+1/v)
    return(result)
def mu_star(psi,w,e,sigma,v,zeta):
    ua=u_a(psi,w,e,sigma,v,zeta)
    ut=u_t(psi,w,e,sigma,v,zeta)
    return(ut-ua)

Evaluating the integrals

To use Gauss-Hermitte quadrature I need to change the support of log-normal distribution from (0,inf) to (-inf,inf). To achieve that I am using change of variables $y=ln(x-\mu)$.
$x=e^{y}+\mu$, $dy/dx=1/(x-\mu)$, $dx=(x-\mu)*dy$. The lognormal distribution after the change in variables transforms to: $\int_{0}^{inf}f(x)dx=\int_{0}^{inf} \frac{1}{x \sigma \sqrt{2 \pi}} e^{-\frac{ln(x-\mu)^2}{2 \sigma^2}}dx=\int_{-inf}^{inf}\frac{x-\mu}{x \sigma \sqrt{2 \pi}} e^{-\frac{y^2}{2 \sigma^2}}dy$

import math
from scipy.stats import norm

def norm_cdf(stddev,mean,x):
    dist=norm(scale=stddev,loc=mean)
    return(dist.cdf(x))
#Lognormal pdf
def lognormal_pdf(stddev,mean,x):
    result=math.exp(-pow(math.log(x)-mean,2)/(2*pow(stddev,2)))/(x*stddev*math.sqrt(2*math.pi))
    return(result)
    #Lognormal pdf transformed. 
def trans_lognormal_pdf(stddev,mean,e):
    #random variable is from infinity to -infinity
    y=e
    x=math.exp(e)+mean
    result=(x-mean)*math.exp(-pow(y,2)/(2*pow(stddev,2)))/(x*stddev*math.sqrt(2*math.pi))
    return(result)

#Gamma pdf
def gamma_pdf(shape,scale,x):
    result=math.exp(-x/scale)*pow(x,shape-1)/(gamma(shape)*pow(scale,shape))
    return(result)

import numpy as np
#Gauss-Hermitte
def integrand_GH(function,e):
    return function(e)*math.exp(pow(e,2))
def integral_GH(function):
    ti, wi = np.polynomial.hermite.hermgauss(150)
    i=0
    result=0
    while i<len(ti):
        result+=wi[i]*integrand_GH(function,ti[i])
        i+=1
    return(result)
#Gauss-Laguerre
def integrand_GL(function,x):
    return function(x)*math.exp(x)

def integral_GL(function):
    ti, wi = np.polynomial.laguerre.laggauss(150)
    i=0
    result=0
    while i<len(ti):
        result+=wi[i]*integrand_GL(function,ti[i])
        i+=1
    return(result)
